3. **In async_setup_entry**, after creating inverter and storing in hass.data, instantiate the optimizer. The optimizer needs coordinator and provider, but those are created in sensor.py's async_setup_entry. To avoid a circular dependency, the optimizer should be created AFTER platforms are set up. However, the timer can start and gracefully handle the first cycle.

Better approach: Create the optimizer eagerly in __init__.py, but the coordinator/provider will be injected by sensor.py after it creates them. So:

```python
async def async_setup_entry(hass, entry) -> bool:
    hass.data.setdefault(DOMAIN, {})
    config = {**entry.data, **entry.options}
    inverter = create_inverter(config["inverter_type"], hass, config)

    hass.data[DOMAIN][entry.entry_id] = {
        "config": config,
        "inverter": inverter,
        # "coordinator", "provider", "optimizer", "select" added by platform setup
    }

    await hass.config_entries.async_forward_entry_setups(entry, PLATFORMS)

    # After platforms are set up, coordinator/provider/select are available
    data = hass.data[DOMAIN][entry.entry_id]
    coordinator = data.get("coordinator")
    provider = data.get("provider")

    if coordinator and provider:
        optimizer = EEGOptimizer(hass, config, inverter, coordinator, provider)
        data["optimizer"] = optimizer

        async def _optimizer_cycle(_now=None):
            select = data.get("select")
            mode = select._attr_current_option if select else MODE_AUS
            if mode != MODE_AUS:
                decision = await optimizer.async_run_cycle(mode)
                # Update decision sensor if available
                sensor = data.get("decision_sensor")
                if sensor and decision:
                    sensor.update_from_decision(decision)

        unsub = async_track_time_interval(
            hass, _optimizer_cycle, timedelta(seconds=60)
        )
        entry.async_on_unload(unsub)

    entry.async_on_unload(entry.add_update_listener(_async_update_listener))
    return True
```
